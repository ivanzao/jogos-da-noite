        function resetChecklist() {
            const now = new Date();
            const resetHour = 3;  0000 GMT-3
            const resetTime = new Date();
            resetTime.setUTCHours(resetHour + 3, 0, 0, 0);
            if (now = resetTime) {
                localStorage.setItem('checklistDate', now.toDateString());
                localStorage.removeItem('checkedItems');
            }
        }

        function loadChecklist() {
            resetChecklist();
            const checkedItems = JSON.parse(localStorage.getItem('checkedItems'))  {};
            document.querySelectorAll('input[type=checkbox]').forEach(checkbox = {
                checkbox.checked = checkedItems[checkbox.id]  false;
                checkbox.addEventListener('change', () = {
                    checkedItems[checkbox.id] = checkbox.checked;
                    localStorage.setItem('checkedItems', JSON.stringify(checkedItems));
                });
            });
        }
